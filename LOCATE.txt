คือการค้นหาตำแหน่งจากรสตริง paramiterที่1 ด้วย paramiterที่2 และสามารถกำหนดตำแหน่งเริ่มต้นที่จะหาได้จาก paramiterที่3
มันจะคืนค่าตำแหน่งที่หาเจอมาให้แต่ถ้าหาเจอมากกว่าหนึ่งมันจะคืนค่าตัวแรก ถ้าหาไม่เจอจะคืนค่ามาเป็น 0

SELECT LOCATE('bar', 'foobarbar');
-> 4

SELECT LOCATE('xbar', 'foobar');
-> 0

SELECT LOCATE('bar', 'foobarbar', 5);
->7
